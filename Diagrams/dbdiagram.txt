Table users {
  username varchar(50) [pk]
  password_hash text [NOT NULL]
  name varchar(50) [NOT NULL]
  elo int [default: 100, NOT NULL]
  coins int [default: 20, NOT NULL]
  wins int [default: 0, NOT NULL]
  losses int [default: 0, NOT NULL]
  bio varchar(200) [default: NULL]
  image varchar(200) [default: NULL]
}

Table user_cards_stack {
  username varchar(50) [ref: > users.username]
  card_id text [ref: > cards.id]

  Indexes {
    (username, card_id) [pk]
  }
}

Table user_cards_deck {
  username varchar(50) [ref: > users.username]
  card_id text [ref: > cards.id]

  Indexes {
    (username, card_id) [pk]
  }
}

Table cards {
  id text [pk]
  name text [NOT NULL]
  type card_type [NOT NULL]
  element card_element [NOT NULL]
  damage float [NOT NULL]
}

Table packs {
  id int [pk, increment]
}

Table pack_cards {
  pack_id int  [ref: > packs.id]
  card_id text [ref: > cards.id]

  Indexes {
    (pack_id, card_id) [pk]
  }
}

Enum card_type {
  Spell
  Goblin
  Troll
  Elf
  Knight
  Dragon
  Ork
  Kraken
  Wizard
}

Enum card_element {
  normal
  water
  fire
}
